* 每次从右侧取一个元素放到最左边，这样每个元素被放到指定位置的概率都相同

```cpp
class Solution {
public:
    Solution(vector<int>& nums) {
        v.assign(begin(nums), end(nums));
    }
    
    /** Resets the array to its original configuration and return it. */
    vector<int> reset() {
        return v;
    }
    
    /** Returns a random shuffling of the array. */
    vector<int> shuffle() {
        vector<int> res = v;
        for (int i = 0; i < size(v); ++i)
        {
            int t = i + rand() % (size(v) - i); // t范围为[i + 1, size(v) - 1]
            swap(res[i], res[t]);
        }
        return res;
    }
private:
    vector<int> v;
};
```

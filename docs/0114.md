* 实现如下

```cpp
class Solution {
public:
    void flatten(TreeNode* root) {
        while (root)
        {
            if (root->left)
            {
                TreeNode* t = root->left;
                while (t->right) t = t->right; // 左子树最右节点
                t->right = root->right;
                root->right = root->left;
                root->left = nullptr;
            }
            root = root->right;
        }
    }
};
```

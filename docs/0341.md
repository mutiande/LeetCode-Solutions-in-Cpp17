* 不断递归直到获取单元素即可

```cpp
class NestedIterator {
public:
    NestedIterator(vector<NestedInteger> &nestedList) {
        append(nestedList, q);
    }
    
    void append(vector<NestedInteger>& nestedList, queue<int>& q)
    {
        for (auto& x : nestedList)
        {
            if (x.isInteger())
            {
                q.emplace(x.getInteger());
            }
            else
            {
                append(x.getList(), q);
            }
        }
    }
    
    int next() {
        int x = q.front();
        q.pop();
        return x;
    }

    bool hasNext() {
        return !empty(q);
    }
private:
    queue<int> q;
};
```

* 求出`0`到`size(nums)`和，再减去数组中所有元素即为结果

```cpp
class Solution {
public:
    int missingNumber(vector<int>& nums) {
        int n = size(nums);
        n = n * (n + 1) / 2;
        return accumulate(begin(nums), end(nums), n, minus<int>{});
    }
};
```

* 如果考虑溢出

```cpp
class Solution {
public:
    int missingNumber(vector<int>& nums) {
        int res = size(nums);
        for (int i = 0; i < size(nums); ++i)
        {
            res += i - nums[i];
        }
        return res;
    }
};
```
